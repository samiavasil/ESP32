Getting Started with ESP-IDF

Created сряда 06 ноември 2019 

1 	mkdir -p ~/Projects/ESP/esp-idf
2 	
3 	cd ~/Projects/ESP/esp-idf
4 	
5 	git clone --recursive https://github.com/espressif/esp-idf.git
6 	
7 	cd esp-idf/
8 	
9 	./install.sh    //Installs tools in ~/.expresive directory
10 	
11 	. ./export.sh   //Export build environment
12 	
13 	cd examples/protocols/websocket
14 	
15 	idf.py menuconfig // Configure application
16 	
17 	idf.py build
18 	
19 	idf.py -p PORT [-b BAUD] flash
20 	
21 	idf.py -p /dev/ttyUSB0 monitor
