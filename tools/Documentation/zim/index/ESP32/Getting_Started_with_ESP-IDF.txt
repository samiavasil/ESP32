====== Getting Started with ESP-IDF ======
Created сряда 06 ноември 2019

{{{code: lang="makefile" linenumbers="True"
	mkdir -p ~/Projects/ESP/esp-idf
	
	cd ~/Projects/ESP/esp-idf
	
	git clone --recursive https://github.com/espressif/esp-idf.git
	
	cd esp-idf/
	
	./install.sh    //Installs tools in ~/.expresive directory
	
	. ./export.sh   //Export build environment
	
	cd examples/protocols/websocket
	
	idf.py menuconfig // Configure application
	
	idf.py build
	
	idf.py -p PORT [-b BAUD] flash
	
	idf.py -p /dev/ttyUSB0 monitor
}}}
